#include <iostream>

using namespace std;
double** matrixx (double **mas,int n1)
{
    setlocale(LC_ALL,"Rus");
    double per;
    double **E=new double *[n1];
    for(int i=0;i<n1;i++)
        E[i]=new double [n1];
    for (int i = 0; i < n1; i++)
    {
        for (int j = 0; j < n1; j++)
        {
            E[i][j] = 0;
        }
        E[i][i] = 1;
    }
    for(int k=0;k<n1 - 1;k++)
    {
        
        for(int i=k+1;i<n1;i++)
        {
            per=mas[i][k] / mas[k][k];
            for(int j=0;j<n1;j++)
            {
                mas[i][j]-=mas[k][j]*(per);
                E[i][j]-=E[k][j]*(per);
            }
        }
    }
    for(int k=n1-1;k>0;k--)
    {
        
        for(int i=k-1;i>=0;i--)
        {
            per=mas[i][k]/mas[k][k];
            for(int j=0;j<n1;j++)
            {
                mas[i][j]-=mas[k][j]*(per);
                E[i][j]-=E[k][j]*(per);
            }
        }
    }
    double m=1;
    for(int i=0;i<n1;i++)
    {
        m*=mas[i][i];
        for(int j=0;j<n1;j++)
        {
            E[i][j]/=mas[i][i];
        }
        mas[i][i]/=mas[i][i];
    }
    cout<<"Определитель матрицы: "<<round(m)<<endl;
    
    for(int i=0;i<n1;i++)
        for(int j=0;j<n1;j++)
            mas[i][j]=E[i][j];
    for(int i=0;i<n1;i++)
        delete [] E[i];
    delete [] E;
    cout<<"Обратная матрица: "<<endl;
    for(int i=0;i<n1;i++)
    {
        for(int j=0;j<n1;j++)
            cout<<mas[i][j]<<setw(5);
        cout<<endl;
    }
    return mas;
}
